for loops

v  = 7:9;
for i = 1:length(v)
	fprintf('%d\n', v(i));
end

count = 0;
for n = rand(1,3)
	count = count + 1;
	fprintf('Element %d ... %f\n', count, n(1));
end

m = rand(3,4);
[rows cols] = size(m);
p = 0;
for row = 1:rows
	for col = 1:cols
		p = p + m(row, col);
	end
end

while loops

total = 0;
n = 0;
while total <= 50
	n = n + 1;
	total = total + n;
end
fprintf('sum: %d, count: %d\n', total, n);
